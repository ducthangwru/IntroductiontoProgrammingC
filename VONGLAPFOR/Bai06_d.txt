S(n) = 1/1! - 1/2! + 1/3! - 1/4! .... 1/n!

#include <stdio.h>
#include <conio.h>

int main()
{
    int n = 0;
    printf ("Nhap n = "); scanf ("%d", &n);

    int i = 0, mau = 1;
    float s = 0;

    for (i = 1; i <= n; i++)
    {
        s += 1.0 / mau;
        mau = -mau * (i+1);
    }

    printf ("S(%d) = %f", n, s);

    _getch();
    return 0;
}
